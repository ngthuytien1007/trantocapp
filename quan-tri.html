Hướng dẫn tích hợp Google Sheet Web API

Giải pháp ghi dữ liệu hai chiều (Read-Write Sync)

Do file CSV đã xuất bản ở chế độ Read-only, ta sử dụng một Google Apps Script làm Web API trung gian để tiếp nhận dữ liệu từ Frontend gửi lên rồi lưu lại vào Sheet.

Bước 1: Cài đặt Apps Script trên Google Sheets

  1. Mở file Google Sheet dữ liệu gia phả dòng họ.
  2. Vào thanh menu: Tiện ích mở rộng (Extensions) > Apps Script.
  3. Xóa toàn bộ mã mặc định và dán đoạn mã Google Apps Script mẫu bên dưới vào.
  4. Bấm vào nút Triển khai (Deploy) > Triển khai mới (New deployment).
  5. Chọn loại cấu hình là Ứng dụng Web (Web app).
  6. Đặt quyền truy cập là Bất kỳ ai (Anyone) (Để Frontend có thể fetch).
  7. Copy lại đường link URL Ứng dụng Web được cấp.

Bước 2: Đoạn mã Google Apps Script mẫu

function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet();
  var membersSheet = sheet.getSheetByName("Members"); // Hoặc tên Tab chứa thành viên
  var newsSheet = sheet.getSheetByName("News");
  var fundsSheet = sheet.getSheetByName("Funds");
  
  var payload = JSON.parse(e.postData.contents);
  var action = payload.action;
  var data = payload.data;
  
  try {
    if (action === "approve_request" || action === "add_member") {
      // Tìm dòng trống cuối và chèn thành viên mới
      membersSheet.appendRow([
        data.id, 
        data.name, 
        data.generation, 
        data.relation, 
        data.dob, 
        data.branch, 
        data.pid, 
        data.motherName, 
        data.isDead ? "YES" : "NO", 
        data.burial || ""
      ]);
    } else if (action === "update_member") {
      // Tìm thành viên theo ID và cập nhật các cột tương ứng
      var rows = membersSheet.getDataRange().getValues();
      for (var i = 1; i < rows.length; i++) {
        if (rows[i][0] === data.id) {
          var rowNum = i + 1;
          membersSheet.getRange(rowNum, 2).setValue(data.name);
          membersSheet.getRange(rowNum, 3).setValue(data.generation);
          membersSheet.getRange(rowNum, 4).setValue(data.relation);
          membersSheet.getRange(rowNum, 5).setValue(data.dob);
          membersSheet.getRange(rowNum, 6).setValue(data.branch);
          membersSheet.getRange(rowNum, 7).setValue(data.pid);
          membersSheet.getRange(rowNum, 8).setValue(data.motherName || "");
          membersSheet.getRange(rowNum, 9).setValue(data.isDead ? "YES" : "NO");
          membersSheet.getRange(rowNum, 10).setValue(data.burial || "");
          break;
        }
      }
    } else if (action === "delete_member") {
      // Tìm thành viên theo ID và xóa dòng tương ứng
      var rows = membersSheet.getDataRange().getValues();
      for (var i = 1; i < rows.length; i++) {
        if (rows[i][0] === data.id) {
          membersSheet.deleteRow(i + 1);
          break;
        }
      }
    } else if (action === "add_news") {
      newsSheet.appendRow([new Date(), data.type, data.title, data.content, data.image || ""]);
    } else if (action === "update_fund") {
      fundsSheet.appendRow([new Date(), data.fundType, data.type, data.amount, data.notes]);
    }
    
    return ContentService.createTextOutput(JSON.stringify({ "status": 200, "message": "Success" }))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch(error) {
    return ContentService.createTextOutput(JSON.stringify({ "status": 500, "message": error.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

Bước 3: Tải lại và xóa Cache (Reload Sync)

Sau khi Frontend nhận được kết quả Status 200 từ Web App Apps Script, hàm Javascript trên Frontend lập tức gọi hàm xóa cache cục bộ (`localStorage.removeItem('trantoc_csv_data')`) và kích hoạt gọi lại bộ nạp dữ liệu để cập nhật bảng giao diện ngay lập tức cho Admin thấy kết quả tức thì.
